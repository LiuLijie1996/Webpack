### 修改webpack.config.js配置中的entry属性

```js
//将打包的入口文件entry属性值改成对象
//多入口打包文件
entry: {
    /*
    * 属性名是可以随便取得,但是最好见名知意
    * 属性名会作为打包后的文件名
    * 这里的路径就是需要打包生成的文件路径
    * */
    aaa:'./src/js/a.js',
    bbb:'./src/js/b.js',
    index:'./src/js/index.js',
},
```

### 修改webpack.config.js配置中的output属性

```js
//打包后的文件的名字和地址
output: {
    //设置打包后的文件统一存放的地址，地址必须是绝对路径
    path: path.join(__dirname, 'dest'),
    //设置打包后的js文件存放的文件夹和名字
    //这里的js文件名不能是规定好的名字，必须使用类似于 [name] 的占位符
    //因为打包后会生成多个js文件，如果名字都一样肯定是不行的
    filename: "js/[name].js",
},
```

### 修改webpack.config.js配置中的plugins属性

```js
    //用来拓展webpack功能的，它们会在整个构建过程中生效，执行相关的任务
    plugins:[
        //将打包好的css代码放在文件中
        new MiniCssExtractPlugin({
            //设置打包后的css文件存放的文件夹和名字
            //因为打包后会有多个css文件，所有文件名肯定不能使用相同的名字的
            //所以使用 [name] 占位符
            filename:"css/[name].css",
        }),

        //压缩打包后的css代码
        new OptimizeCssAssetsPlugin(),

        //使用多少次HtmlWebpackPlugin插件就会生成多少个html文件
        new HtmlWebpackPlugin({
            //模板文件
            template: "./src/template.html",
            //打包后文件的名字
            filename: "aaa.html",
            //规定打包后的文件只需要引入aaa.js以及和aaa.js文件相关的文件
            chunks:['aaa'],
            /*
            * 模板中是可以写ejs代码的
            * 我们可以自定义一些属性,如title
            * 在模板中通过 <%= htmlWebpackPlugin.options.title %>
            * 即可在打包时引用属性值
            * */
            title:"我是aaa.html文件"
        }),

        new HtmlWebpackPlugin({
            //模板文件
            template: "./src/template.html",
            //打包后文件的名字
            filename: "bbb.html",
            //规定打包后的文件只需要引入bbb.js以及和bbb.js文件相关的文件
            chunks:['bbb'],
            title:"我是bbb.html文件"
        }),
    ]
```
